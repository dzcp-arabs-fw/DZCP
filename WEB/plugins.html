Creating a Basic Plugin

To create a plugin in DZCP, follow this example:

using DZCP.Bootstrap;
using DZCP.Core;

public class MyPlugin : IPlugin
{
    public void OnLoad()
    {
        Console.WriteLine("[MyPlugin] Plugin loaded successfully!");
    }
}
